Morphological Analysis

The study of word structure and formation, breaking words into morphemes (roots, prefixes, suffixes)

Key Techniques

  • Stemming - reducing words to root forms
  • Lemmatization - reducing words to dictionary forms with context
  • Morphological parsing - identifying word components
  • Word formation analysis

Code Examples

import nltk
from nltk.stem import PorterStemmer

stemmer = PorterStemmer()
words = ['running', 'runs', 'easily', 'fairly']
stemmed = [stemmer.stem(word) for word in words]
print(stemmed)  # ['run', 'run', 'easili', 'fairli']
import spacy

nlp = spacy.load('en_core_web_sm')
doc = nlp('better mice running quickly')
lemmas = [token.lemma_ for token in doc]
print(lemmas)  # ['good', 'mouse', 'run', 'quickly']

Lexical Analysis

Processing text to identify and categorize words (tokens) and assign grammatical roles

Key Techniques

  • Tokenization - splitting text into words/tokens
  • Part-of-Speech (POS) tagging - assigning grammatical categories
  • Stop word removal - filtering common words
  • Named Entity Recognition - identifying proper nouns

Pragmatic Analysis

Understanding intended meaning beyond literal interpretation, considering context and speaker intent

Key Techniques

  • Intent recognition - understanding speaker's purpose
  • Context analysis - using surrounding information
  • Implicature detection - finding implied meanings
  • Speech act recognition - identifying communication goals

# Simple intent recognition example
import re

def analyze_intent(text):
    text = text.lower()
    
    # Request patterns
    if re.search(r'can you|could you|would you', text):
        return "REQUEST"
    elif re.search(r'what|where|when|how|why', text):
        return "QUESTION"
    elif re.search(r'hello|hi|hey', text):
        return "GREETING"
    else:
        return "STATEMENT"

examples = [
    "Can you pass the salt?",
    "What time is it?", 
    "Hello there!",
    "It's cold in here"
]

for example in examples:
    intent = analyze_intent(example)
    print(f"'{example}' -> {intent}")
# Context-based pragmatic analysis
def analyze_pragmatics(text, context=None):
    analysis = {
        'literal_meaning': text,
        'pragmatic_meaning': None,
        'speech_act': None
    }
    
    text_lower = text.lower()
    
    # Analyze speech acts
    if '?' in text:
        if text_lower.startswith(('can you', 'could you', 'would you')):
            analysis['speech_act'] = 'REQUEST'
            analysis['pragmatic_meaning'] = 'Polite request for action'
        else:
            analysis['speech_act'] = 'QUESTION'
            analysis['pragmatic_meaning'] = 'Seeking information'
    elif text_lower.startswith(('it\'s', 'this is')):
        analysis['speech_act'] = 'ASSERTION'
        if 'cold' in text_lower or 'hot' in text_lower:
            analysis['pragmatic_meaning'] = 'Indirect request to adjust temperature'
    
    return analysis

Discourse Integration

Understanding how multiple sentences form coherent discourse, analyzing relationships between sentences and maintaining context across larger text units

Key Techniques

  • Coreference resolution - identifying expressions that refer to the same entity
  • Discourse segmentation - breaking text into coherent discourse segments
  • Coherence analysis - measuring how well sentences stick together
  • Anaphora resolution - linking pronouns to their antecedents
  • Discourse parsing - identifying discourse structure relationships
  • Topic segmentation - dividing text into topic-based segments

# Discourse coherence analysis using sentence embeddings
from sentence_transformers import SentenceTransformer
import numpy as np
from scipy.spatial.distance import cosine

def analyze_discourse_coherence(text):
    # Initialize sentence transformer
    model = SentenceTransformer('all-MiniLM-L6-v2')
    
    # Split into sentences
    sentences = text.split('.')
    sentences = [s.strip() for s in sentences if s.strip()]
    
    # Get sentence embeddings
    embeddings = model.encode(sentences)
    
    # Calculate coherence scores between adjacent sentences
    coherence_scores = []
    for i in range(len(embeddings) - 1):
        similarity = 1 - cosine(embeddings[i], embeddings[i + 1])
        coherence_scores.append(similarity)
    
    return {
        'sentences': sentences,
        'coherence_scores': coherence_scores,
        'average_coherence': np.mean(coherence_scores)
    }

# Example usage
text = "John loves programming. He codes every day. Programming is his passion."
result = analyze_discourse_coherence(text)
print(f"Average coherence: {result['average_coherence']:.3f}")
# Simple coreference detection using rule-based approach
import re
from collections import defaultdict

def simple_coreference_detection(text):
    # Simple patterns for coreference
    entities = {}
    coref_chains = defaultdict(list)
    
    # Find named entities (simplified)
    entity_pattern = r'\b[A-Z][a-z]+\b'  # Simple proper noun detection
    entities_found = re.findall(entity_pattern, text)
    
    # Find pronouns and their positions
    pronoun_pattern = r'\b(he|she|it|they|him|her|them|his|hers|its|their)\b'
    
    # Simple resolution: link pronouns to nearest preceding entity
    sentences = text.split('.')
    current_entity = None
    
    for i, sentence in enumerate(sentences):
        sentence = sentence.strip()
        if not sentence:
            continue
            
        # Check for entities in this sentence
        for entity in entities_found:
            if entity in sentence:
                current_entity = entity
                coref_chains[entity].append(f"Sentence {i+1}: {entity}")
                break
        
        # Check for pronouns
        pronouns_in_sentence = re.findall(pronoun_pattern, sentence.lower())
        for pronoun in pronouns_in_sentence:
            if current_entity:
                coref_chains[current_entity].append(f"Sentence {i+1}: {pronoun}")
    
    return coref_chains

# Example
text = "John went to the store. He bought milk and bread. His groceries were expensive."
chains = simple_coreference_detection(text)
for entity, mentions in chains.items():
    print(f"{entity}: {mentions}")
